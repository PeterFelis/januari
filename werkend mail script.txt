<?php

ini_set('display_errors', 1);
ini_set('display_startup_errors', 1);
error_reporting(E_ALL)

use PHPMailer\PHPMailer\PHPMailer;
use PHPMailer\PHPMailer\Exception;

require 'PHPMailer/src/Exception.php';
require 'PHPMailer/src/PHPMailer.php';
require 'PHPMailer/src/SMTP.php';

$mail = new PHPMailer(true);

try {
    // Gebruik SMTP
    $mail->isSMTP();
    $mail->Host = 'mail225.hostingdiscounter.nl'; // SMTP-server van Hostingdiscounter
    $mail->SMTPAuth = true;
    $mail->Username = 'info@fetum.nl'; // vervang dit met jouw e-mailadres
    $mail->Password = 'rNqjQ2h4EC';       // vervang dit met jouw wachtwoord
    $mail->SMTPSecure = PHPMailer::ENCRYPTION_STARTTLS; // Gebruik TLS, anders probeer PHPMailer::ENCRYPTION_SMTPS voor SSL
    $mail->Port = 587; // Poort 587 voor TLS, of 465 voor SSL

    // Stel de afzender en ontvanger in
    $mail->setFrom('info@fetum.nl', 'Fetum');
    $mail->addAddress('peter@felis.nl', 'peter');

    // E-mail inhoud
    $mail->isHTML(false);
    $mail->Subject = 'Test e-mail via Hostingdiscounter SMTP';
    $mail->Body    = "Dit is een testbericht verstuurd via de SMTP-server van Hostingdiscounter.";

    $mail->send();
    echo 'Het bericht is verzonden';
} catch (Exception $e) {
    echo "Er is een fout opgetreden: {$mail->ErrorInfo}";
}
